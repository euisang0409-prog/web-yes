# 1. 문자열 함수

-- CONCAT : 여러 문자열을 하나로 합침 (NULL 있으면 결과도 NULL)
select CONCAT('안녕',null,'하세요');

-- CONCAT_WS : 구분자를 기준으로 문자열 합침 (NULL 자동 무시)
-- 형식: CONCAT_WS('구분자', 값1, 값2 ...)
select concat_ws('-','A',null,'B');

-- NULL 전파 규칙
-- NULL이 포함된 연산은 결과도 NULL
select 10 + null; -- 결과: NULL
select 'A' + NULL; -- 결과: NULL

-- IFNULL : NULL이면 다른 값으로 치환
-- 형식: IFNULL(값, 대체값)
select CONCAT(
NAME,
' -> ',
CHAR_LENGTH(NAME),
IFNULL(NICKNAME,' 없음'),
' -> ',
IFNULL(CHAR_LENGTH(NICKNAME),'0')
)
from MEMBER;

-- SUBSTRING_INDEX : 구분자로 나눈 문자열에서 앞/뒤 일부 추출
-- 형식: SUBSTRING_INDEX(문자열, 구분자, 개수)
select SUBSTRING_INDEX('A-B-C-D','-',3);

# 2. 숫자 함수

-- ABS : 절대값 반환 (음수 -> 양수)
select ABS(-10), ABS(10), ABS(0);

-- ROUND : 반올림
-- ROUND(숫자, 자릿수)
select
ROUND(1234.567,2), -- 소수 2자리
ROUND(1234.567,0), -- 정수 반올림
ROUND(1234.567), -- 기본 (0)
ROUND(1234.567,-2); -- 100단위 반올림

-- FLOOR : 내림 (작거나 같은 최대 정수)
select FLOOR(2), FLOOR(2.1), FLOOR(-2.1);

-- TRUNCATE : 소수점 이하 버림
select TRUNCATE(3.141592,2);

-- CEIL : 올림 (크거나 같은 최소 정수)
select CEIL(2), CEIL(2.1), CEIL(-2.1);

-- SIGN : 부호 반환
-- 양수=1, 음수=-1, 0=0
select SIGN(-238), SIGN(0), SIGN(359);

-- MOD : 나머지 계산
select MOD(1,3), MOD(2,3), MOD(3,3), MOD(4,3);

-- POWER : 거듭제곱
select POWER(2,1), POWER(2,2), POWER(2,3), POWER(2,0);

-- RAND : 0~1 사이 난수
select RAND();

-- GREATEST : 가장 큰 값 반환
-- LEAST : 가장 작은 값 반환
select GREATEST(10,20,5), LEAST(10,20,5);

# 3. sales 테이블

CREATE TABLE sales (
id INT PRIMARY KEY,
product VARCHAR(50),
price DECIMAL(10,2),
quantity INT,
discount_rate DECIMAL(5,2)
);

INSERT INTO sales VALUES
(1, 'Keyboard', 29900, 2, 0.10),
(2, 'Mouse', 15900, 3, 0.05),
(3, 'Monitor', 199000, 1, 0.20),
(4, 'USB', 8900, 10, 0.00),
(5, 'Speaker', 45500, 4, 0.15);

-- MOD : 나머지 계산
select MOD(price,3) from sales;

-- CONCAT + ROUND : 할인율 % 변환
select ID, PRODUCT, CONCAT(ROUND(DISCOUNT_RATE\*100), '%') from sales;

-- 총 가격 계산 (price _ quantity)
select id, product, price, quantity, price _ quantity from sales;

-- RAND : 1~100 난수 생성
select (RAND()\*100) + 1;

-- SIGN : 기준값 대비 비교 (-1,0,1)
select product, SIGN(price - 50000) from sales;

-- FORMAT : 숫자를 문자열 + 콤마 형태로 변환
select FORMAT(1234567,0), FORMAT(1234567,2);

select id, product, FORMAT(price,0), quantity, discount_rate from sales;

# 4. 날짜 함수

-- NOW : 현재 날짜 + 시간
select NOW();

-- CURDATE : 현재 날짜
select CURDATE();

-- CURTIME : 현재 시간
select CURTIME();

-- YEAR : 연도 추출
select YEAR('2026-04-30');

-- MONTH : 월 추출
select MONTH('2026-04-30');

-- DAY : 일 추출
select DAY('2026-04-30');

-- HOUR / MINUTE / SECOND : 시간 분 초 추출
select HOUR('2026-04-30 15:20:10'),
MINUTE('2026-04-30 15:20:10'),
SECOND('2026-04-30 15:20:10');

-- DAYOFWEEK : 요일 숫자 (1=일요일)
select DAYOFWEEK('2026-04-30');

-- DAYNAME : 요일 이름
select DAYNAME(CURDATE());

-- LAST_DAY : 해당 월 마지막 날
select CURDATE(), LAST_DAY(CURDATE());

-- DATE_ADD : 날짜 더하기
select NOW(),
DATE_ADD(NOW(), INTERVAL 10 DAY),
DATE_ADD(NOW(), INTERVAL 3 MONTH),
DATE_ADD(NOW(), INTERVAL 2 HOUR);

-- DATE_SUB : 날짜 빼기
select NOW(),
DATE_SUB(NOW(), INTERVAL 10 DAY),
DATE_SUB(NOW(), INTERVAL 3 MONTH),
DATE_SUB(NOW(), INTERVAL 2 HOUR);

-- DATEDIFF : 날짜 차이 (일 단위)
select DATEDIFF('2026-05-10','2026-04-30');

-- DATE_FORMAT : 날짜 포맷 변경
select DATE_FORMAT('2026-04-30 15:20:10','%Y년 %m월 %d일 %H시 %i분');

-- MAKEDATE : 연도 + 일수로 날짜 생성
select MAKEDATE(2026,32);
