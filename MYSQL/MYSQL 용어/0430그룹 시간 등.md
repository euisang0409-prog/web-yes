# 1. 문자열 함수

select CONCAT('안녕',null,'하세요');

-- CONCAT_WS : 구분자 포함 문자열 합치기
select concat_ws('-','A',null,'B');

-- NULL 전파 규칙
-- NULL이 포함되면 연산 결과도 NULL
select 10 + null; -- NULL
select 'A' + NULL;

-- IFNULL : NULL 값 대체
select CONCAT(
NAME,
' -> ',
CHAR_LENGTH(NAME),
IFNULL(NICKNAME,' 없음'),
' -> ',
IFNULL(CHAR_LENGTH(NICKNAME),'0')
)
from MEMBER;

-- SUBSTRING_INDEX
select SUBSTRING_INDEX('A-B-C-D','-',3);

# 3. sales 테이블

CREATE TABLE sales (
id INT PRIMARY KEY,
product VARCHAR(50),
price DECIMAL(10,2),
quantity INT,
discount_rate DECIMAL(5,2)
);

INSERT INTO sales VALUES
(1, 'Keyboard', 29900, 2, 0.10),
(2, 'Mouse', 15900, 3, 0.05),
(3, 'Monitor', 199000, 1, 0.20),
(4, 'USB', 8900, 10, 0.00),
(5, 'Speaker', 45500, 4, 0.15);

-- 가격 나머지
select MOD(price,3) from sales;

-- 할인율 % 표시
select ID, PRODUCT, CONCAT(ROUND(DISCOUNT_RATE*100), '%') from sales;

-- 총 가격
select id, product, price, quantity, price * quantity from sales;

-- 난수
select (RAND()*100) + 1;

-- 가격 비교
select product, SIGN(price - 50000) from sales;

-- FORMAT
select FORMAT(1234567,0), FORMAT(1234567,2);

select id, product, FORMAT(price,0), quantity, discount_rate from sales;

# 5. orders 테이블

CREATE TABLE orders (
id INT PRIMARY KEY AUTO_INCREMENT,
customer VARCHAR(50),
order_date DATE,
ship_date DATE,
price INT
);

INSERT INTO orders (customer, order_date, ship_date, price) VALUES
('홍길동', '2025-01-02', '2025-01-05', 30000),
('이몽룡', '2025-01-10', '2025-01-12', 45000),
('성춘향', '2024-12-28', '2025-01-03', 52000),
('김철수', '2025-02-01', '2025-02-03', 15000),
('박영희', '2025-02-10', NULL, 22000);

-- 배송일 계산 (NULL 제외)
SELECT
customer,
DATEDIFF(ship_date, order_date)
FROM orders
WHERE ship_date IS NOT NULL;

-- 2025년 1월 주문
SELECT *
FROM orders
WHERE YEAR(order_date) = 2025
AND MONTH(order_date) = 1;

-- 작년 주문
SELECT *
FROM orders
WHERE YEAR(order_date) = YEAR(CURDATE()) - 1;

# 6. employees 테이블

DROP TABLE IF EXISTS employees;

CREATE TABLE employees (
employee_id INT PRIMARY KEY,
first_name VARCHAR(50),
last_name VARCHAR(50),
email VARCHAR(100),
hire_date DATE,
job_id VARCHAR(20),
salary INT,
commission_pct DECIMAL(3,2),
manager_id INT,
department_id INT
);

-- 데이터 생략 (기존 동일)

-- 이름 마스킹
SELECT REPLACE(first_name,'el','**')
FROM employees
WHERE department_id = 30;

-- 이름 길이
SELECT first_name, salary
FROM employees
WHERE CHAR_LENGTH(first_name) >= 6;

-- 짝수/홀수
SELECT employee_id, first_name, MOD(employee_id,2)
FROM employees;

-- 짝수만
SELECT employee_id, first_name
FROM employees
WHERE MOD(employee_id,2) = 0;

-- 막대 그래프
SELECT first_name, salary, REPEAT('■', ROUND(salary / 1000))
FROM employees;

-- 6개월 후
SELECT first_name, hire_date, DATE_ADD(hire_date, INTERVAL 6 MONTH)
FROM employees;

# 7. 집계 함수

SELECT COUNT(*) FROM employees;
SELECT SUM(salary), AVG(salary), MAX(salary), MIN(salary) FROM employees;

-- IT_PROG 통계
SELECT AVG(salary), MAX(salary), MIN(salary), SUM(salary)
FROM employees
WHERE job_id = 'IT_PROG';

-- 평균 반올림
SELECT ROUND(AVG(salary),1)
FROM employees
WHERE employee_id = 100;

-- 2005 입사자
SELECT COUNT(*)
FROM employees
WHERE YEAR(hire_date) = 2005;

# 8. GROUP BY

SELECT job_id, COUNT(*)
FROM employees
GROUP BY job_id;

SELECT job_id, SUM(salary)
FROM employees
GROUP BY job_id;

SELECT department_id, MAX(salary)
FROM employees
GROUP BY department_id;

-- 부서 + 직무 그룹
SELECT department_id, job_id, COUNT(*)
FROM employees
GROUP BY department_id, job_id;

-- 부서별 평균 (5000 이상)
SELECT department_id, AVG(salary)
FROM employees
WHERE salary >= 5000
GROUP BY department_id;

-- 급여 차이
SELECT job_id, MAX(salary) - MIN(salary) AS salary_gap
FROM employees
GROUP BY job_id;

-- 이름에 a 포함 + 이름 길이별
SELECT CHAR_LENGTH(first_name), COUNT(*)
FROM employees
WHERE first_name LIKE '%a%'
GROUP BY CHAR_LENGTH(first_name);
