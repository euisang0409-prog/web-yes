# SQL 문자열 함수 정리 (MySQL)

## 1. 문자열 길이

```sql
-- length() -> 몇바이트 인지 반환
SELECT title, LENGTH(title) FROM film_practice;

-- 테이블에 저장된 데이터가 아니라, 값 자체를 함수에 직접 전달해서
-- 결과를 확인할 수 있다.
SELECT LENGTH('가나다');

-- char_length()
-- 글자 수를 반환한다.
-- 한글 이모지를 포함해서 실제 글자 수를 보고싶을 때 사용한다.
SELECT CHAR_LENGTH('가나다');
SELECT title, CHAR_LENGTH(title) FROM film_practice;

-- select 절에 사용하는 것은 조회해서 확인을 하겠다.

-- 영화제목이 10자 이하인 것만 영화제목(title)조회하세요
SELECT title, CHAR_LENGTH(title)
FROM film_practice fp
WHERE CHAR_LENGTH(title) <= 10;
```

## 2. 대소문자 변환

```sql
-- Upper(), UCASE()
-- 문자열을 대문자로 변환
SELECT UPPER('hello MySQL');

-- 검색에서 대소문자 무시하고 비교하고 싶을 때 사용한다.

-- LOWER()
-- 모든 글자를 소문자로 출력

-- CUSTOMER 테이블의 이름을 소문자로 조회하기
SELECT FIRST_NAME, LOWER(first_name)
FROM customer_practice;

-- 테이블에 들어있는 데이터가 대문자인지 소문자인지 모를 때
-- 전부다 대문자로 바꾸던지, 소문자로 변환해서 사용
```

## 3. 공백 제거

```sql
-- TRIM(), LTRIM(), RTRIM()
-- 공백제거하기
SELECT TRIM('      HI      '); -- 양쪽 공백 제거
SELECT LTRIM('      HI      '); -- 왼쪽 공백 제거
SELECT RTRIM('      HI      '); -- 오른쪽 공백 제거

-- 특정 문자 지우기 문법 제공
SELECT TRIM('X' FROM 'xxxHELLOxxx');
```

## 4. LPAD / RPAD

```sql
-- LPAD(), RPAD()
-- 왼쪽/오른쪽에 문자열을 채워 넣는 함수
-- LPAD(문자열, 총길이, 채울문자)

SELECT LPAD('HELLO',10,'*');
SELECT RPAD('HELLO',10,'*');

-- 문자열이 길이를 초과하게 되면 자른다.
SELECT LPAD('ABCDEFcase',4,'0');

-- 고객테이블에서 회원번호의 자리수를 5로 만들고
-- 빈자리는 0으로 채워 고객 번호와 이름을 조회하세요
SELECT LPAD(customer_id,5,0),first_name
FROM customer_practice cp;
```

## 5. 문자열 자르기

```sql
-- SUNSTRING(문자열,시작위치,길이)
-- SUBSTR()
-- MID()

SELECT SUBSTRING('hello MySQL', 1,5);
-- 글자를 셀 때 1부터 시작한다.

-- 길이를 생략하면 끝까지 자른다.
SELECT SUBSTR('Hello MySQL',7);

-- 음수 사용 가능함(뒤에서부터 자른다.)
SELECT MID('Hello MySQL',-5);

-- LEFT(), RIGHT()
-- 문자열에서 왼쪽/오른쪽에서 원하는 길이만큼 잘라서 반환하는 함수

SELECT 
	LEFT('Hello', 2),
	right('Hello',2);

-- '900101-1234567' 문자열에서 주민번호 앞 6자리만 자르기
SELECT LEFT('900101-1234567',6);

-- photo.png에서 파일 확장자 추출하기
SELECT RIGHT('photo.png',4);
```

## 6. INSTR

```sql
-- INSTR(전체 문자열, 찾을 문자열)
-- 찾을 문자열이 시작되는 위치번호를 반환
-- 위치번호는 1부터 시작
-- 찾지 못하면 0을 반환한다.

SELECT
	INSTR('HELLO','L'),
	INSTR('사과바나나포도','바나나'),
	INSTR('사과바나나포도','딸기');

-- 이메일에서 @위치 찾기
-- 'user01@TEST.COM'
SELECT INSTR('user01@TEST.COM','@');
```

## 7. 이메일 아이디 추출

```sql
-- 이메일에서 아이디 부분을 추출하기
SELECT
	MID(
	'user01@test.com',
	1,
	INSTR('user01@TEST.COM','@')-1);

-- customer_practice
SELECT
	customer_id,
	MID(email,1,INSTR(email,'@')-1)
FROM customer_practice cp;

-- 고객테이블에서 이메일에 @sakilacustomer.org 포함 고객 조회
SELECT customer_id,first_name
FROM customer_practice cp
WHERE INSTR(email, '@sakilacustomer.org') > 0;
```

## 8. CONCAT

```sql
-- CONCAT('문자열1','문자열',...)
-- 문자열 이어붙이기

SELECT
	CONCAT('HELLO','WORLD'),
	CONCAT('HELLO',' ','WORLD');

SELECT CONCAT('나이 : ',25,'세');

-- 고객 테이블에서 이름과 성 연결
SELECT CONCAT(first_name,' ', last_name)
FROM customer_practice cp;
```

## 9. REPLACE

```sql
-- REPLACE(문자열, OLD, NEW)
SELECT REPLACE('ABCDAB','AB','XY');

-- 공백 문자 치환
SELECT REPLACE('hello world', ' ', '_');

-- 이메일 도메인 변경
CREATE TABLE member (
  id INT PRIMARY KEY,
  name VARCHAR(50),
  email VARCHAR(100),
  phone VARCHAR(50),
  nickname VARCHAR(50)
);

INSERT INTO member VALUES
(1, '홍길동', 'gildong@test.com', '010-1234-5678', '길동이'),
(2, '김영희', 'younghee@test.com', '010-999-8888', '영희짱'),
(3, '이철수', 'chulsoo@oldmail.com', '010-7777-3333', '철수'),
(4, 'Jane', 'jane@example.com', '010-1212-3434', NULL);

SELECT * FROM member;

-- 이메일 도메인 변경
SELECT REPLACE(EMAIL,'@test.com','@school.com') FROM member;
```

## 10. REPEAT

```sql
-- REPEAT(문자열, 횟수)
SELECT REPEAT('ABC',3);
SELECT REPEAT('*',5);

-- 이름 반복
SELECT REPEAT('홍길동',3);
```

## 11. 실전 문제

```sql
-- 이름 + 이메일
SELECT CONCAT(name,'-',email)
FROM member;

-- 전화번호 하이픈 제거
SELECT REPLACE(PHONE, '-','')
FROM member;

-- 이메일 도메인 변경
SELECT REPLACE(EMAIL,'@test.com','@school.com') FROM member;

-- 전화번호 마스킹
SELECT CONCAT(left(phone,3),'-****-****')
FROM member;

-- 이름 글자수
SELECT name,char_length(name),char_length(nickname)
FROM member;

-- 국내번호 아님
SELECT CONCAT('국내번호 아님 : ', phone)
FROM member
WHERE LEFT(phone,3) <> '010';

-- 이름 첫글자 마스킹
SELECT CONCAT(
	LEFT(first_name,1),
	REPEAT('*',char_length(first_name)-1)
)
FROM customer_practice cp;

-- film A로 시작
SELECT *
FROM film_practice fp
WHERE LEFT(title,1) = 'A';
```
